# 개요
개별 댓글을 표시하는 컴포넌트로, 확장/축소 상태를 props로 받아 처리

# Design Pattern
- **Presentational Component**: 상태를 직접 관리하지 않고 Props로 받아 렌더링.
- **Controlled Component**: 확장 상태(`isExpanded`)를 부모가 제어.

# Input / Output

## Input
Props {
    nametag: string,
    content: string,
    commentId: string,
    isExpanded: boolean,
    onExpand: (id: string) => void
}

## Output
Output {
    expandRequest: string  // commentId
}

# Pseudo-code
import color_template.css
import markdown     // 프레임워크 별 적절한 markdown 라이브러리
import katex        // 프레임워크 별 적절한 katex 라이브러리

VerticalGrid(
    elements: [
        nametag,
        Padding(vLength: 8px),
        render(content, (markdown, katex)) if isExpanded else content,
    ],
    border: Property(
        color: color_template.grey-lv2,
        width: 1px,
        radius: 0px,
        ),
    fill: Property(
        color: color_template.background
        ) if isExpanded else Property(
        color: color_template.grey-lv1 if hover == True else color_template.background,
    ),
    hWidth: max,
    vLength: isExpanded ? "auto" : 96px,  // 확장 시 자동 높이
    innerShadow: Property(null) if isExpanded else Property(
        color: color_template.background,
        x-offset: 0px,
        y-offset: -8px
    ),
    
    onPress: (event) => {
        event.stopPropagation()  // 이벤트 버블링 방지
        onExpand(commentId)  // 부모에게 확장 요청
    }
)
