# 개요

추천페이지에 표시되는 추천 콘텐츠 컴포넌트.

# Design Pattern
- **Presentational Component**: 데이터를 받아 단순히 렌더링하는 역할.
- **Event Delegation**: 클릭 이벤트를 상위로 전달.

# Input / Output

## Input
Props {
    title: string,
    description: string,
    favicon: string,
    url: string,
    keyword: list[string],
    onPageClick: (url: string) => void
}

## Output
Output {
    onPageClick: (url: string) => void  // 페이지 클릭 시 URL 전달
}

# Pseudo code

import color_template.css

## component

Box(
    element: 
        VerticalGrid(
            elements: [
                HorizontalGrid(elements: [RenderFavicon(favicon), Text(title, fontColor: color_template.font-black)]),
                HorizontalLine(color: color_template.grey-lv3, width: 1px),
                Text(description, fontColor: color_template.font-black),
                HorizontalGrid(elements: [Text("#" + k, fontColor: color_template.grey-lv3) for k in keyword]),
            ]
        ),
    fill: {
        borderColor: color_template.grey-lv2,
        borderWidth: 2px,
        borderRadius: 0,
        hover: color_template.grey-lv1,
    },
    onPress: () => {
        Props.onPageClick(Props.url)
    }
)
